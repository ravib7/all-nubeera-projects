                                                        
                                                             When Button Click Read Span Value in Div


--------------------------------------------------------------------------------------------------------------------------------------------------------------
                                                                        HTML
-------------------------------------------------------------------------------------------------------------------------------------------------------------


<!DOCTYPE html>
<head>
    <link rel="stylesheet" href="practice.css">
</head>

<body>

 <div id="myDiv">
    <span id="mySpan">Hello World !</span>
 </div>

<br>

 <button id="myButton">Check div value</button>

    <script src="practice.js"></script>
</body>


-------------------------------------------------------------------------------------------------------------------------------------------------------------
                                                                              CSS
-------------------------------------------------------------------------------------------------------------------------------------------------------------


#myDiv {
    padding: 7px;
    padding-right: 10px;
    border: 1px solid black;
}


--------------------------------------------------------------------------------------------------------------------------------------------------------------
                                                                            JS
--------------------------------------------------------------------------------------------------------------------------------------------------------------

var myButton = document.getElementById("myButton");
myButton.addEventListener("click", function() {
    var myDiv = document.getElementById("myDiv");
    var mySpan = myDiv.querySelector("#mySpan");
    var  mySpanValue = mySpan.innerHTML;
    alert(mySpanValue);
});
